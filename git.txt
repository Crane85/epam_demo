 git config --global user.name - проверка прописанного имени

 git config --global user.email - проверка прописанного емейла

 cd /c/users/AZhuravlev/onedrive/java/ - вход в директорию репозитория

 git clone git@github.com:Crane85/epam_demo.git - синхронизация локальной директории и репозитория

 ls - проверка синхронизации

 git remote -v - проверка подлиности репозитория

 git status - проверка изменений репозиторий/директория

 git add new.txt - добавление изменений

 git reset - откат обновлений

 git commit - "одобрение" обновления (-m " " - сообщение в коммите)
	--amend -m - дописание коммита

 git log -  история коммитов

 git push -  синхронизация репозитория и директории

 git pull - синхронизация и подтягивание изменений

 git gui& - открытие графического интерфейса

 gitk& - просмотр всех изменений репозитория

 git show - показывает последние изменения
	bc4506 - показывает содержимое выбранного файла
	-s --pretty=raw первые цифры коммита - показывает детальную информацию о коммите

 git ls-tree первые цифры sh1 дерева показывает blob содержимого дерева

 git checkout - откат изменений до индекса и коммита (необратимо)
	-- <file> - откат изменений в выбраном файле
	. - откат всех изменений

 git clean - удаление новых файлов до индекса и коммита
	-x - игнорирование правила ignore
	-d - удаление новых директорий
	-f - форсирование (запуск, без него не рОбит)

 git reset --<file> - перевод файла в неиндес
	HEAD^ (HEAD~1) откат коммита ^ или ~1 количество коммитов, которые надо откатить
	--soft - откат коммита в индекс
	--mixed - откат коммита в доиндекс (можно без него, это дефолтное значение)
	--hard - удаление коммита

 git revert <sha1> - удаление с репозитория (можно ревертить реверты)

 .gitignore - создается файл в директории и в него прописываются или файлы, или расширения файлов, которые будут игнорироваться при коммитах и пушах
	*.log - игнорируются все лог файлы
		! - делает исключение из исключений
	/ - игнорируются директории
	<dir>/ - игнорируются директории в выбранной директории
	<dir>/*.txt - игнорируются все файлы txt в выбранной директории
	<dir>/**/*.pdf - игнорируются все файлы pdf во всех дочерних директориях выбранной директории

 git checkout - работа с ветками
	-b <name> - создание новой ветки (бранч и имя ветки)
	-master - переход в ветку мастер
	-feature - переход в ветку фичер
	-abort - отмена слияния

 git merge feature - слияние веток (делается из ветки, в которую надо произвести слияние)
